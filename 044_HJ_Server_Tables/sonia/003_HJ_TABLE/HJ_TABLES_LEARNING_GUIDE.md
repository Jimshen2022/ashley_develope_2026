### **??2: ???????PPH**
```sql
SELECT 
  t.team_id,
  t.team_name,
  COUNT(DISTINCT tm.employee_id) AS EmployeeCount,
  SUM(tl.tran_qty) AS TotalQty,
  DATEDIFF(HOUR, MIN(tc.start_time), MAX(tc.end_time)) AS WorkHours
FROM t_team t (NOLOCK)
INNER JOIN t_team_member tm (NOLOCK) ON t.wh_id = tm.wh_id AND t.team_id = tm.team_id
INNER JOIN t_team_employee_clock_in_out tc (NOLOCK) ON tm.wh_id = tc.wh_id AND tm.employee_id = tc.employee_id
LEFT JOIN t_tran_log tl (NOLOCK) ON tm.wh_id = tl.wh_id AND tm.employee_id = tl.employee_id
WHERE t.wh_id = '1'
  AND t.group_id = 'S'
  AND tc.work_day = CAST(GETDATE() AS DATE)
GROUP BY t.team_id, t.team_name
```

### **??3: ??????**
```sql
SELECT 
  l.location_id,
  l.area,
  l.aisle,
  l.section,
  l.tier,
  COUNT(si.serial_number) AS SerialCount,
  SUM(si.actual_qty) AS TotalQty
FROM t_location l (NOLOCK)
LEFT JOIN t_stored_item si (NOLOCK) ON l.wh_id = si.wh_id AND l.location_id = si.location_id
WHERE l.wh_id = '1'
```

```sql
SELECT TOP 100
```
